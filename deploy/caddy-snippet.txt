# Caddy site block for canquery.
#
# The Express process serves the built SPA itself in production, so this is a
# pure reverse_proxy - no static file mounts needed. Substitute <your-domain>.
#
# The proxy target is the API's loopback address. If Caddy runs inside Docker
# and the API runs on the host, target the Docker bridge gateway instead
# (commonly 172.x.0.1) so the container can reach the host's :3100.

<your-domain> {
	encode zstd gzip
	header -Server
	reverse_proxy 127.0.0.1:3100
}

www.<your-domain> {
	redir https://<your-domain>{uri} permanent
}
